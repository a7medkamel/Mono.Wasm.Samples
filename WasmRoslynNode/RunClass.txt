using System.Text;
namespace WasmRoslyn.Demo
{
    public class RunClass
    {
        public string Run(params object[] args)
        {
            var name = args[0];
            var count = (int)args[1];
            var sb = new StringBuilder();
            for (int i = 0; i < count; i++)
            {
                sb.AppendLine($"{i}) Hello, {name}!");
            }
            return sb.ToString();
        }
    }
}
